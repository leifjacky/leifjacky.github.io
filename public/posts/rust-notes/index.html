  fn calc(wf: f64) -> f64 {
      wf + 127.3
  }

  fn fibonaci(mut n: i8) -> i64 {
      if n == 1 {
          return 1;
      } else if n == 2 {
          return 2;
      }
      let mut last = 2;
      let mut ret = 3;
      while n > 3 {
          let temp = ret;
          ret += last;
          last = temp;
          n -= 1;
      }
      return ret;
  }
